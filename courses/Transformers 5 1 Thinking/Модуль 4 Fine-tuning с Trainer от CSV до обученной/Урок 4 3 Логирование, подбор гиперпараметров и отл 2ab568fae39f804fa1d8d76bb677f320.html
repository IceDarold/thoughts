import numpy as np
from sklearn.metrics import accuracy_score, precision_recall_fscore_support

from transformers import (
    AutoModelForSequenceClassification,
    TrainingArguments,
    Trainer,
)
from data_pipeline import load_and_prepare_data


def compute_metrics(eval_pred):
    logits, labels = eval_pred

    preds = np.argmax(logits, axis=-1)

    acc = accuracy_score(labels, preds)
    precision, recall, f1, _ = precision_recall_fscore_support(
        labels,
        preds,
        average="weighted",
        zero_division=0,
    )

    return {
        "accuracy": acc,
        "f1": f1,
        "precision": precision,
        "recall": recall,
    }


def main():
    csv_path = "tickets.csv"
    base_model_name = "distilbert-base-uncased"

    # 1. –î–∞–Ω–Ω—ã–µ
    (
        tokenized_dataset,
        label2id,
        id2label,
        tokenizer,
        data_collator,
    ) = load_and_prepare_data(
        csv_path=csv_path,
        model_name=base_model_name,
        max_length=128,
        test_size=0.1,
        seed=42,
    )

    # 2. –ú–æ–¥–µ–ª—å
    num_labels = len(label2id)
    model = AutoModelForSequenceClassification.from_pretrained(
        base_model_name,
        num_labels=num_labels,
        id2label=id2label,
        label2id=label2id,
    )

    # 3. –ê—Ä–≥—É–º–µ–Ω—Ç—ã –æ–±—É—á–µ–Ω–∏—è —Å –±–æ–ª–µ–µ —Ç–æ–Ω–∫–∏–º–∏ –Ω–∞—Å—Ç—Ä–æ–π–∫–∞–º–∏
    training_args = TrainingArguments(
        output_dir="checkpoints/tickets_tuned",

        # === Eval / Save ===
        evaluation_strategy="epoch",   # eval –≤ –∫–æ–Ω—Ü–µ –∫–∞–∂–¥–æ–π —ç–ø–æ—Ö–∏
        save_strategy="epoch",         # —á–µ–∫–ø–æ–∏–Ω—Ç –≤ –∫–æ–Ω—Ü–µ –∫–∞–∂–¥–æ–π —ç–ø–æ—Ö–∏
        save_total_limit=3,            # –Ω–µ —Ö—Ä–∞–Ω–∏—Ç—å –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ –º–Ω–æ–≥–æ —á–µ–∫–ø–æ–∏–Ω—Ç–æ–≤

        # === –û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è ===

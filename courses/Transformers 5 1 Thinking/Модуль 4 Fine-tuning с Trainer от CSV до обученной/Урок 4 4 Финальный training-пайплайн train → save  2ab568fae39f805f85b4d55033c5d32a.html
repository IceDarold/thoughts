# train_and_infer.py

import numpy as np
from sklearn.metrics import accuracy_score, precision_recall_fscore_support

from transformers import (
    AutoModelForSequenceClassification,
    AutoTokenizer,
    TrainingArguments,
    Trainer,
)

from data_pipeline import load_and_prepare_data


def compute_metrics(eval_pred):
    """
    –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Ä–∞—Å—á—ë—Ç–∞ –º–µ—Ç—Ä–∏–∫ –Ω–∞ –≤–∞–ª–∏–¥–∞—Ü–∏–∏.
    Trainer –≤—ã–∑—ã–≤–∞–µ—Ç –µ—ë –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –ø—Ä–∏ evaluate().
    """
    logits, labels = eval_pred  # EvalPrediction —Ä–∞—Å–ø–∞–∫–æ–≤—ã–≤–∞–µ—Ç—Å—è –≤ (predictions, label_ids)

    # –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω—ã–π –∫–ª–∞—Å—Å ‚Äî argmax –ø–æ –ª–æ–≥–∏—Ç–∞–º
    preds = np.argmax(logits, axis=-1)

    acc = accuracy_score(labels, preds)
    precision, recall, f1, _ = precision_recall_fscore_support(
        labels,
        preds,
        average="weighted",
        zero_division=0,
    )

    return {
        "accuracy": acc,
        "f1": f1,
        "precision": precision,
        "recall": recall,
    }


def train_model(
    csv_path: str,
    base_model_name: str = "distilbert-base-uncased",
    output_dir: str = "checkpoints/tickets_final",
    num_epochs: int = 4,
    learning_rate: float = 3e-5,
    max_length: int = 128,
    test_size: float = 0.1,
    seed: int = 42,
) -> str:
    """
    –û–±—É—á–∞–µ—Ç –º–æ–¥–µ–ª—å –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏ —Ç–∏–∫–µ—Ç–æ–≤ –Ω–∞ –¥–∞–Ω–Ω—ã—Ö –∏–∑ csv_path.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø—É—Ç—å –∫ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏ —Å —Å–æ—Ö—Ä–∞–Ω—ë–Ω–Ω–æ–π —Ñ–∏–Ω–∞–ª—å–Ω–æ–π –º–æ–¥–µ–ª—å—é.
    """

    # 1. –ì–æ—Ç–æ–≤–∏–º –¥–∞–Ω–Ω—ã–µ
    (
        tokenized_dataset,
        label2id,
        id2label,
        tokenizer,
        data_collator,
    ) = load_and_prepare_data(
        csv_path=csv_path,
        model_name=base_model_name,
        max_length=max_length,
        test_size=test_size,
        seed=seed,
    )

    # 2. –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º –º–æ–¥–µ–ª—å
    num_labels = len(label2id)
    model = AutoModelForSequenceClassification.from_pretrained(
        base_model_name,
        num_labels=num_labels,
        id2label=id2label,
        label2id=label2id,
    )

    # 3. –ù–∞—Å—Ç—Ä–∞–∏–≤–∞–µ–º TrainingArguments
    training_args = TrainingArguments(
        output_dir=output_dir,

        # eval / save
        evaluation_strategy="epoch",
        save_strategy="epoch",
        save_total_limit=3,

        # –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è
        learning_rate=learning_rate,
        per_device_train_batch_size=16,
        per_device_eval_batch_size=32,
        num_train_epochs=num_epochs,
        weight_decay=0.01,
        warmup_ratio=0.1,

        # –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ
        logging_strategy="steps",
        logging_steps=20,
        logging_first_step=True,
        report_to=["none"],  # –º–æ–∂–Ω–æ –ø–æ–º–µ–Ω—è—Ç—å –Ω–∞ ["tensorboard"]

        # –ª—É—á—à–∞—è –º–æ–¥–µ–ª—å
        load_best_model_at_end=True,
        metric_for_best_model="f1",
        greater_is_better=True,
    )

    # 4. –°–æ–∑–¥–∞—ë–º Trainer
    trainer = Trainer(
        model=model,
        args=training_args,
        train_dataset=tokenized_dataset["train"],
        eval_dataset=tokenized_dataset["validation"],
        tokenizer=tokenizer,
        data_collator=data_collator,
        compute_metrics=compute_metrics,
    )

    # 5. –û–±—É—á–∞–µ–º –º–æ–¥–µ–ª—å
    trainer.train()

    # 6. –§–∏–Ω–∞–ª—å–Ω–∞—è –æ—Ü–µ–Ω–∫–∞
    metrics = trainer.evaluate()
    print("Final eval metrics:", metrics)

    # 7. –°–æ—Ö—Ä–∞–Ω—è–µ–º –ª—É—á—à—É—é –º–æ–¥–µ–ª—å –≤ –æ—Ç–¥–µ–ª—å–Ω—É—é –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é
    final_model_dir = "models/tickets_classifier_final"
    trainer.save_model(final_model_dir)
    tokenizer.save_pretrained(final_model_dir)

    print("Best model checkpoint:", trainer.state.best_model_checkpoint)
    print("Final model saved to:", final_model_dir)

    return final_model_dir


def predict_single(text: str, model_dir: str):
    """
    –î–µ–ª–∞–µ—Ç –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ –¥–ª—è –æ–¥–Ω–æ–≥–æ —Ç–µ–∫—Å—Ç–∞ —Ç–∏–∫–µ—Ç–∞.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ª–æ–≤–∞—Ä—å —Å –ø–æ–ª—è–º–∏: text, label, confidence.
    """
    # 1. –ó–∞–≥—Ä—É–∂–∞–µ–º —Ç–æ–∫–µ–Ω–∏–∑–∞—Ç–æ—Ä –∏ –º–æ–¥–µ–ª—å
    tokenizer = AutoTokenizer.from_pretrained(model_dir)
    model = AutoModelForSequenceClassification.from_pretrained(model_dir)

    # 2. –¢–æ–∫–µ–Ω–∏–∑–∏—Ä—É–µ–º —Ç–µ–∫—Å—Ç
    enc = tokenizer(
        text,
        return_tensors="pt",
        padding=True,
        truncation=True,
        max_length=128,
    )

    # 3. –ü—Ä–æ–≥–æ–Ω —á–µ—Ä–µ–∑ –º–æ–¥–µ–ª—å
    import torch
    model.eval()
    with torch.no_grad():
        outputs = model(**enc)

    logits = outputs.logits  # [1, num_labels]
    probs = torch.softmax(logits, dim=-1)[0]  # [num_labels]

    pred_id = torch.argmax(probs).item()
    confidence = probs[pred_id].item()

    id2label = model.config.id2label
    label = id2label[pred_id]

    return {
        "text": text,
        "label": label,
        "confidence": confidence,
    }


def predict_batch(texts, model_dir: str):
    """
    –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –¥–ª—è —Å–ø–∏—Å–∫–∞ —Ç–µ–∫—Å—Ç–æ–≤.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ø–∏—Å–æ–∫ —Å–ª–æ–≤–∞—Ä–µ–π, –∞–Ω–∞–ª–æ–≥–∏—á–Ω—ã—Ö predict_single.
    """
    tokenizer = AutoTokenizer.from_pretrained(model_dir)
    model = AutoModelForSequenceClassification.from_pretrained(model_dir)

    import torch
    enc = tokenizer(
        texts,
        return_tensors="pt",
        padding=True,
        truncation=True,
        max_length=128,
    )

    model.eval()
    with torch.no_grad():
        outputs = model(**enc)

    logits = outputs.logits  # [batch, num_labels]
    probs = torch.softmax(logits, dim=-1)  # [batch, num_labels]
    pred_ids = torch.argmax(probs, dim=-1)  # [batch]

    id2label = model.config.id2label

    results = []
    for text, pred_id, p in zip(texts, pred_ids, probs):
        pred_id = pred_id.item()
        label = id2label[pred_id]
        confidence = p[pred_id].item()
        results.append(
            {
                "text": text,
                "label": label,
                "confidence": confidence,
            }
        )

    return results

    precision, recall, f1, _ = precision_recall_fscore_support(

    return {

def predict_batch(texts, model_dir: str):
    tokenizer = AutoTokenizer.from_pretrained(model_dir)
    model = AutoModelForSequenceClassification.from_pretrained(model_dir)

    import torch
    enc = tokenizer(
        texts,
        return_tensors="pt",
        padding=True,
        truncation=True,
        max_length=128,
    )

    model.eval()
    with torch.no_grad():
        outputs = model(**enc)

    logits = outputs.logits
    probs = torch.softmax(logits, dim=-1)
    pred_ids = torch.argmax(probs, dim=-1)

    id2label = model.config.id2label

    results = []
    for text, pred_id, p in zip(texts, pred_ids, probs):
        pred_id = pred_id.item()
        label = id2label[pred_id]
        confidence = p[pred_id].item()
        results.append(
            {
                "text": text,
                "label": label,
                "confidence": confidence,
            }
        )

    return results

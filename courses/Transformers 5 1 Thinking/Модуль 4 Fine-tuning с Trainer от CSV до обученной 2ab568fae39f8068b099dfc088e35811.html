import numpy as np
from sklearn.metrics import accuracy_score, f1_score, precision_recall_fscore_support
from transformers import (
    AutoModelForSequenceClassification,
    TrainingArguments,
    Trainer,
)
from data_pipeline import load_and_prepare_data


def compute_metrics(eval_pred):
    logits, labels = eval_pred
    # logits shape: [num_samples, num_labels]
    preds = np.argmax(logits, axis=-1)

    acc = accuracy_score(labels, preds)
    precision, recall, f1, _ = precision_recall_fscore_support(
        labels, preds, average="weighted", zero_division=0
    )

    return {
        "accuracy": acc,
        "f1": f1,
        "precision": precision,
        "recall": recall,
    }


def main():
    csv_path = "tickets.csv"
    model_name = "distilbert-base-uncased"

    tokenized_dataset, label2id, id2label, tokenizer, data_collator = load_and_prepare_data(
        csv_path=csv_path,
        model_name=model_name,
        max_length=128,
        test_size=0.1,
        seed=42,
    )

    num_labels = len(label2id)
    model = AutoModelForSequenceClassification.from_pretrained(
        model_name,
        num_labels=num_labels,
        id2label=id2label,
        label2id=label2id,
    )

    training_args = TrainingArguments(
        output_dir="checkpoints/tickets_metrics",
        evaluation_strategy="epoch",
        save_strategy="epoch",
        learning_rate=2e-5,
        per_device_train_batch_size=16,
        per_device_eval_batch_size=32,
        num_train_epochs=3,
        weight_decay=0.01,
        logging_steps=50,
        load_best_model_at_end=True,
        metric_for_best_model="f1",
        greater_is_better=True,
    )

    trainer = Trainer(
        model=model,
        args=training_args,
        train_dataset=tokenized_dataset["train"],
        eval_dataset=tokenized_dataset["validation"],
        tokenizer=tokenizer,
        data_collator=data_collator,
        compute_metrics=compute_metrics,
    )

    trainer.train()

    metrics = trainer.evaluate()
    print("Final eval metrics:", metrics)

    trainer.save_model("models/tickets_classifier_f1")
    tokenizer.save_pretrained("models/tickets_classifier_f1")


if __name__ == "__main__":
    main()

import numpy as np
from sklearn.metrics import accuracy_score, f1_score, precision_recall_fscore_support

from transformers import (
    AutoModelForSequenceClassification,
    AutoTokenizer,
    TrainingArguments,
    Trainer,
)

from data_pipeline import load_and_prepare_data


def compute_metrics(eval_pred):
    logits, labels = eval_pred
    preds = np.argmax(logits, axis=-1)

    acc = accuracy_score(labels, preds)
    precision, recall, f1, _ = precision_recall_fscore_support(
        labels, preds, average="weighted", zero_division=0
    )

    return {
        "accuracy": acc,
        "f1": f1,
        "precision": precision,
        "recall": recall,
    }


def train_model(
    csv_path: str,
    base_model_name: str = "distilbert-base-uncased",
    output_dir: str = "checkpoints/tickets_final",
    num_epochs: int = 4,
    learning_rate: float = 3e-5,
):
    # 1. –î–∞–Ω–Ω—ã–µ
    tokenized_dataset, label2id, id2label, tokenizer, data_collator = load_and_prepare_data(
        csv_path=csv_path,
        model_name=base_model_name,
        max_length=128,
        test_size=0.1,
        seed=42,
    )

    # 2. –ú–æ–¥–µ–ª—å
    num_labels = len(label2id)
    model = AutoModelForSequenceClassification.from_pretrained(
        base_model_name,
        num_labels=num_labels,
        id2label=id2label,
        label2id=label2id,
    )

    # 3. –ê—Ä–≥—É–º–µ–Ω—Ç—ã –æ–±—É—á–µ–Ω–∏—è
    training_args = TrainingArguments(
        output_dir=output_dir,
        evaluation_strategy="epoch",
        save_strategy="epoch",
        learning_rate=learning_rate,
        per_device_train_batch_size=16,
        per_device_eval_batch_size=32,
        num_train_epochs=num_epochs,
        weight_decay=0.01,
        warmup_ratio=0.1,
        logging_steps=20,
        load_best_model_at_end=True,
        metric_for_best_model="f1",
        greater_is_better=True,
        report_to=["none"],  # –º–æ–∂–Ω–æ –ø–æ–º–µ–Ω—è—Ç—å –Ω–∞ ["tensorboard"]
    )

    # 4. Trainer
    trainer = Trainer(
        model=model,
        args=training_args,
        train_dataset=tokenized_dataset["train"],
        eval_dataset=tokenized_dataset["validation"],
        tokenizer=tokenizer,
        data_collator=data_collator,
        compute_metrics=compute_metrics,
    )

    # 5. –û–±—É—á–µ–Ω–∏–µ
    trainer.train()

    # 6. –§–∏–Ω–∞–ª—å–Ω–∞—è –æ—Ü–µ–Ω–∫–∞
    metrics = trainer.evaluate()
    print("Final metrics:", metrics)

    # 7. –°–æ—Ö—Ä–∞–Ω—è–µ–º –ª—É—á—à—É—é –º–æ–¥–µ–ª—å –≤ –æ—Ç–¥–µ–ª—å–Ω—É—é –ø–∞–ø–∫—É (–¥–ª—è –ø—Ä–æ–¥–∞–∫—à–µ–Ω–∞)
    final_model_dir = "models/tickets_classifier_final"
    trainer.save_model(final_model_dir)
    tokenizer.save_pretrained(final_model_dir)

    return final_model_dir


def predict_single(text: str, model_dir: str):
    # –ò–Ω—Ñ–µ—Ä–µ–Ω—Å –ø–æ—Å–ª–µ –æ–±—É—á–µ–Ω–∏—è
    tokenizer = AutoTokenizer.from_pretrained(model_dir)
    model = AutoModelForSequenceClassification.from_pretrained(model_dir)

    enc = tokenizer(
        text,
        return_tensors="pt",
        padding=True,
        truncation=True,
        max_length=128,
    )

    import torch
    with torch.no_grad():
        outputs = model(**enc)

    logits = outputs.logits
    probs = torch.softmax(logits, dim=-1)[0]
    pred_id = torch.argmax(probs).item()

    id2label = model.config.id2label
    label = id2label[pred_id]
    confidence = probs[pred_id].item()

    return {
        "text": text,
        "label": label,
        "confidence": confidence,
    }


if __name__ == "__main__":

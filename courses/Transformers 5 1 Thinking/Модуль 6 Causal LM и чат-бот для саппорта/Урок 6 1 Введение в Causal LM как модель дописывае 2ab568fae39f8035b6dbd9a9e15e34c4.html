import torch
from transformers import AutoTokenizer, AutoModelForCausalLM

def main():
    model_name = "gpt2"  # –º–æ–∂–Ω–æ –∑–∞–º–µ–Ω–∏—Ç—å –Ω–∞ –¥—Ä—É–≥—É—é causal LM –º–æ–¥–µ–ª—å

    # 1. –ó–∞–≥—Ä—É–∂–∞–µ–º —Ç–æ–∫–µ–Ω–∏–∑–∞—Ç–æ—Ä –∏ –º–æ–¥–µ–ª—å
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    model = AutoModelForCausalLM.from_pretrained(model_name)

    # –í GPT-2 –Ω–µ—Ç pad_token ‚Äî –ø–æ–¥—Å—Ç—Ä–∞—Ö—É–µ–º—Å—è:
    if tokenizer.pad_token is None:
        tokenizer.pad_token = tokenizer.eos_token

    # 2. –ü—Ä–∏–¥—É–º—ã–≤–∞–µ–º –ø—Ä–æ–º–ø—Ç: –¥–∏–∞–ª–æ–≥ user + –ø—Ä–∏–≥–ª–∞—à–µ–Ω–∏–µ –æ—Ç–≤–µ—Ç–∞
    prompt = (
        "System: You are a helpful and polite customer support agent.\n"
        "User: The app started crashing after the last update when I open the gallery.\n"
        "Support:"
    )

    # 3. –¢–æ–∫–µ–Ω–∏–∑–∏—Ä—É–µ–º –ø—Ä–æ–º–ø—Ç
    inputs = tokenizer(
        prompt,
        return_tensors="pt",
    )

    print("Input shape:", inputs["input_ids"].shape)

    # 4. –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –ø—Ä–æ–¥–æ–ª–∂–µ–Ω–∏–µ
    with torch.no_grad():
        output_ids = model.generate(
            **inputs,
            max_new_tokens=60,          # —Å–∫–æ–ª—å–∫–æ –ù–û–í–´–• —Ç–æ–∫–µ–Ω–æ–≤ —Ö–æ—Ç–∏–º —Å–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞—Ç—å
            do_sample=True,             # –≤–∫–ª—é—á–∞–µ–º —Å—Ç–æ—Ö–∞—Å—Ç–∏—á–µ—Å–∫—É—é –≥–µ–Ω–µ—Ä–∞—Ü–∏—é
            top_p=0.9,                  # nucleus sampling
            temperature=0.7,            # —á—É—Ç—å —Å–≥–ª–∞–∂–∏–≤–∞–µ–º —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ
            repetition_penalty=1.1,     # —à—Ç—Ä–∞—Ñ –∑–∞ –ø–æ–≤—Ç–æ—Ä
            eos_token_id=tokenizer.eos_token_id,  # –∫–æ–≥–¥–∞ –º–æ–∂–Ω–æ –æ—Å—Ç–∞–Ω–æ–≤–∏—Ç—å—Å—è
        )

    print("Output shape:", output_ids.shape)

    # 5. –î–µ–∫–æ–¥–∏—Ä—É–µ–º –ø–æ–ª–Ω–æ—Ç–µ–∫—Å—Ç–æ–≤—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç
    full_text = tokenizer.decode(output_ids[0], skip_special_tokens=True)
    print("=== FULL TEXT ===")
    print(full_text)

    # 6. –í—ã—Ç–∞—Å–∫–∏–≤–∞–µ–º —Ç–æ–ª—å–∫–æ ‚Äú–æ—Ç–≤–µ—Ç —Å–∞–ø–ø–æ—Ä—Ç–∞‚Äù (—Ö–≤–æ—Å—Ç)
    input_len = inputs["input_ids"].shape[1]
    new_tokens = output_ids[0][input_len:]   # —Ç–æ–ª—å–∫–æ —Å–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞–Ω–Ω–∞—è —á–∞—Å—Ç—å

    print("\n=== SUPPORT REPLY ONLY ===")
    print(reply.strip())


if __name__ == "__main__":

def generate_reply(
    model,
    tokenizer,
    user_message: str,
    do_sample: bool = True,
    num_beams: int | None = None,
    top_p: float = 0.9,
    temperature: float = 0.7,
    max_new_tokens: int = 60,
):
    prompt = (
        "System: You are a helpful and concise support assistant.\n"
        f"User: {user_message}\n"
        "Support:"
    )

    inputs = tokenizer(
        prompt,
        return_tensors="pt",
    )

    gen_kwargs = dict(
        max_new_tokens=max_new_tokens,
        eos_token_id=tokenizer.eos_token_id,
        repetition_penalty=1.1,
    )

    if num_beams is not None:
        # beam search
        gen_kwargs.update(
            num_beams=num_beams,
            do_sample=False,
        )
    else:
        # sampling / greedy
        gen_kwargs.update(
            do_sample=do_sample,
            top_p=top_p,
            temperature=temperature,
        )

    with torch.no_grad():
        output_ids = model.generate(
            **inputs,
            **gen_kwargs,
        )

    input_len = inputs["input_ids"].shape[1]
    new_tokens = output_ids[0][input_len:]
    reply = tokenizer.decode(new_tokens, skip_special_tokens=True)

    return reply.strip()

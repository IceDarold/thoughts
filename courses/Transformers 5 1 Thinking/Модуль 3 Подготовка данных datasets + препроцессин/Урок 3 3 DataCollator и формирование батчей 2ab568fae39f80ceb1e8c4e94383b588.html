from collections import Counter

from datasets import load_dataset, DatasetDict
from transformers import AutoTokenizer, DataCollatorWithPadding, AutoModelForSequenceClassification
from torch.utils.data import DataLoader
import torch


def main():
    # === 1. –ó–∞–≥—Ä—É–∂–∞–µ–º –∏ —Å–ø–ª–∏—Ç–∏–º CSV ===
    raw = load_dataset("csv", data_files="tickets.csv")["train"]

    split = raw.train_test_split(test_size=0.1, seed=42)
    dataset = DatasetDict(
        {
            "train": split["train"],
            "validation": split["test"],
        }
    )

    print("=== Raw DatasetDict ===")
    print(dataset)

    # === 2. –°—Ç—Ä–æ–∏–º label_list, label2id, id2label ===
    unique_labels = sorted(set(dataset["train"]["label"]))
    label_list = unique_labels
    label2id = {label: i for i, label in enumerate(label_list)}
    id2label = {i: label for label, i in label2id.items()}

    print("\n=== Labels ===")
    print("label_list:", label_list)
    print("label2id:", label2id)

    print("\n=== Label distribution (train) ===")
    print(Counter(dataset["train"]["label"]))

    # === 3. –¢–æ–∫–µ–Ω–∏–∑–∞—Ç–æ—Ä ===
    model_name = "distilbert-base-uncased"
    tokenizer = AutoTokenizer.from_pretrained(model_name)

    # === 4. –§—É–Ω–∫—Ü–∏—è –ø—Ä–µ–ø—Ä–æ—Ü–µ—Å—Å–∏–Ω–≥–∞ ===
    def preprocess_function(batch):
        texts = batch["text"]

        tokenized = tokenizer(
            texts,
            truncation=True,
            max_length=128,
            # padding=False, –ø–∞–¥–¥–∏–Ω–≥ —Å–¥–µ–ª–∞–µ–º collator'–æ–º
        )

        labels = [label2id[l] for l in batch["label"]]
        tokenized["labels"] = labels

        return tokenized

    # === 5. –¢–æ–∫–µ–Ω–∏–∑–∏—Ä—É–µ–º dataset —á–µ—Ä–µ–∑ map ===
    tokenized_dataset = dataset.map(
        preprocess_function,
        batched=True,
        remove_columns=["text", "label"],
    )

    print("\n=== Tokenized DatasetDict ===")
    print(tokenized_dataset)

    # === 6. DataCollatorWithPadding ===
    data_collator = DataCollatorWithPadding(
        tokenizer=tokenizer,
        return_tensors="pt",
    )

    # –ü—Ä–æ–≤–µ—Ä–∏–º, —á—Ç–æ collator –¥–µ–ª–∞–µ—Ç
    samples = [tokenized_dataset["train"][i] for i in range(3)]

    print("\n=== Raw samples (different lengths) ===")
    for s in samples:
        print(
            "len(input_ids)=",
            len(s["input_ids"]),
            "| label=",
            s["labels"],
        )

    batch = data_collator(samples)

    print("\n=== After collator (batched) ===")
    for k, v in batch.items():
        print(k, v.shape, v.dtype)

    # === 7. DataLoader –¥–ª—è train/validation ===
    train_dataloader = DataLoader(
        tokenized_dataset["train"],
        batch_size=8,
        shuffle=True,
        collate_fn=data_collator,
    )

    val_dataloader = DataLoader(
        tokenized_dataset["validation"],
        batch_size=8,
        shuffle=False,
        collate_fn=data_collator,
    )

    first_batch = next(iter(train_dataloader))
    print("\n=== One batch from train_dataloader ===")
    for k, v in first_batch.items():
        print(k, v.shape, v.dtype)

    # === 8. (–ú–∏–Ω–∏-–ø—Ä–æ–≤–µ—Ä–∫–∞) –ü—Ä–æ–≥–æ–Ω—è–µ–º –±–∞—Ç—á —á–µ—Ä–µ–∑ –º–æ–¥–µ–ª—å ===
    model = AutoModelForSequenceClassification.from_pretrained(
        model_name,
        num_labels=len(label2id),
        id2label=id2label,
        label2id=label2id,
    )

    with torch.no_grad():
        outputs = model(
            input_ids=first_batch["input_ids"],
            attention_mask=first_batch["attention_mask"],
            labels=first_batch["labels"],
        )

    print("\n=== Model outputs ===")
    print("Loss:", outputs.loss.item())
    print("Logits shape:", outputs.logits.shape)


if __name__ == "__main__":
    main()

from typing import Tuple, Dict

from datasets import load_dataset, DatasetDict
from transformers import AutoTokenizer, DataCollatorWithPadding


def load_and_prepare_data(
    csv_path: str,
    model_name: str = "distilbert-base-uncased",
    text_column: str = "text",
    label_column: str = "label",
    test_size: float = 0.1,
    max_length: int = 128,
    seed: int = 42,
) -> Tuple[DatasetDict, Dict[str, int], Dict[int, str], AutoTokenizer, DataCollatorWithPadding]:
    """
    –ü–æ–ª–Ω—ã–π data-pipeline:
    CSV -> DatasetDict(train/validation) -> —Ç–æ–∫–µ–Ω–∏–∑–∞—Ü–∏—è -> labels -> DataCollator.

    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
        tokenized_dataset: DatasetDict —Å –ø–æ–ª—è–º–∏ input_ids, attention_mask, labels
        label2id: —Å–ª–æ–≤–∞—Ä—å {"bug": 0, "feature": 1, ...}
        id2label: —Å–ª–æ–≤–∞—Ä—å {0: "bug", 1: "feature", ...}
        tokenizer: AutoTokenizer, —Å–≤—è–∑–∞–Ω–Ω—ã–π —Å model_name
        data_collator: DataCollatorWithPadding –¥–ª—è —ç—Ç–æ–≥–æ —Ç–æ–∫–µ–Ω–∏–∑–∞—Ç–æ—Ä–∞
    """

    # 1. –ó–∞–≥—Ä—É–∂–∞–µ–º CSV
    raw = load_dataset("csv", data_files=csv_path)["train"]

    # 2. train/validation split
    split = raw.train_test_split(test_size=test_size, seed=seed)

    dataset = DatasetDict(
        {
            "train": split["train"],
            "validation": split["test"],
        }
    )

    # 3. –°—Ç—Ä–æ–∏–º label_list, label2id, id2label
    unique_labels = sorted(set(dataset["train"][label_column]))
    label_list = unique_labels

    label2id: Dict[str, int] = {label: i for i, label in enumerate(label_list)}
    id2label: Dict[int, str] = {i: label for label, i in label2id.items()}

    # 4. –¢–æ–∫–µ–Ω–∏–∑–∞—Ç–æ—Ä
    tokenizer = AutoTokenizer.from_pretrained(model_name)

    # 5. –§—É–Ω–∫—Ü–∏—è –ø—Ä–µ–ø—Ä–æ—Ü–µ—Å—Å–∏–Ω–≥–∞
    def preprocess_function(batch):
        texts = batch[text_column]

        tokenized = tokenizer(
            texts,
            truncation=True,
            max_length=max_length,
            # padding=False, –ø–∞–¥–¥–∏–Ω–≥ —Å–¥–µ–ª–∞–µ—Ç collator
        )

        labels = [label2id[l] for l in batch[label_column]]
        tokenized["labels"] = labels

        return tokenized

    # 6. –ü—Ä–∏–º–µ–Ω—è–µ–º map
    tokenized_dataset = dataset.map(
        preprocess_function,
        batched=True,
        remove_columns=[text_column, label_column],
    )

    # 7. DataCollatorWithPadding
    data_collator = DataCollatorWithPadding(
        tokenizer=tokenizer,
        return_tensors="pt",
    )

    # 8. –í–æ–∑–≤—Ä–∞—â–∞–µ–º –≤—Å—ë –¥–æ–±—Ä–æ
    return tokenized_dataset, label2id, id2label, tokenizer, data_collator

    tokenized_dataset, label2id, id2label, tokenizer, data_collator = load_and_prepare_data(
        csv_path="tickets.csv",
        model_name="distilbert-base-uncased",
        text_column="text",
        label_column="label",
        test_size=0.1,
        max_length=128,
        seed=42,
    )

    print("=== Tokenized Dataset ===")
    print(tokenized_dataset)

    print("\n=== Labels mapping ===")
    print("label2id:", label2id)
    print("id2label:", id2label)

    # –ü–æ—Å–º–æ—Ç—Ä–∏–º –Ω–∞ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –º–µ—Ç–æ–∫ –≤ train
    train_labels = [id2label[l] for l in tokenized_dataset["train"]["labels"]]
    print("\n=== Label distribution (train) ===")
    print(Counter(train_labels))

    # 2. –°–æ–±–∏—Ä–∞–µ–º DataLoader
    train_dataloader = DataLoader(
        tokenized_dataset["train"],
        batch_size=8,
        shuffle=True,
        collate_fn=data_collator,
    )

    # –ë–µ—Ä—ë–º –æ–¥–∏–Ω –±–∞—Ç—á
    batch = next(iter(train_dataloader))

    print("\n=== One batch ===")
    for k, v in batch.items():
        print(k, v.shape, v.dtype)

    # 3. –ü–æ–¥–∫–ª—é—á–∞–µ–º –º–æ–¥–µ–ª—å –∏ –ø—Ä–æ–≤–µ—Ä—è–µ–º forward
    model_name = "distilbert-base-uncased"
    model = AutoModelForSequenceClassification.from_pretrained(
        model_name,
        num_labels=len(label2id),
        id2label=id2label,
        label2id=label2id,
    )

    model.eval()

    with torch.no_grad():
        outputs = model(
            input_ids=batch["input_ids"],
            attention_mask=batch["attention_mask"],
            labels=batch["labels"],
        )

    print("\n=== Model outputs ===")
    print("Loss:", outputs.loss.item())
    print("Logits shape:", outputs.logits.shape)  # [batch_size, num_labels]


if __name__ == "__main__":
    main()

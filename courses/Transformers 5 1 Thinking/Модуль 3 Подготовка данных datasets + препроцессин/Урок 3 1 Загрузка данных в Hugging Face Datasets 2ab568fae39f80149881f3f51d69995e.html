from collections import Counter

from datasets import load_dataset, DatasetDict


def main():
    # 1. –ó–∞–≥—Ä—É–∂–∞–µ–º CSV –≤ DatasetDict
    data = load_dataset(
        "csv",
        data_files="tickets.csv",
    )
    train_ds = data["train"]

    print("=== Raw dataset ===")
    print(train_ds)
    print("Example[0]:", train_ds[0])

    # 2. –†–∞–∑–º–µ—Ä –∏ –∫–æ–ª–æ–Ω–∫–∏
    print("\n=== Basic info ===")
    print("Num rows:", len(train_ds))
    print("Features:", train_ds.features)

    # 3. –†–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –∫–ª–∞—Å—Å–æ–≤
    print("\n=== Label distribution (full) ===")
    full_labels = train_ds["label"]
    print(Counter(full_labels))

    # 4. train/validation split
    split = train_ds.train_test_split(
        test_size=0.1,
        seed=42,
    )

    dataset = DatasetDict(
        {
            "train": split["train"],
            "validation": split["test"],
        }
    )

    print("\n=== Split dataset ===")
    print(dataset)
    print("Train size:", len(dataset["train"]))
    print("Validation size:", len(dataset["validation"]))

    # 5. –†–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –∫–ª–∞—Å—Å–æ–≤ –ø–æ —Å–ø–ª–∏—Ç–∞–º
    print("\n=== Label distribution (train) ===")
    train_labels = dataset["train"]["label"]
    print(Counter(train_labels))

    print("\n=== Label distribution (validation) ===")
    val_labels = dataset["validation"]["label"]
    print(Counter(val_labels))

    # 6. –ü—Ä–∏–º–µ—Ä –ø–æ—Å–ª–µ —Å–ø–ª–∏—Ç–∞
    print("\nExample train[0]:", dataset["train"][0])
    print("Example val[0]:", dataset["validation"][0])


if __name__ == "__main__":
    main()

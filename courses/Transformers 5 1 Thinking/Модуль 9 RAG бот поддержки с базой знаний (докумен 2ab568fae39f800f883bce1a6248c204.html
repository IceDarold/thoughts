import torch
from transformers import AutoTokenizer, AutoModel

model_id = "sentence-transformers/all-MiniLM-L6-v2"  # –æ–Ω –∂–µ –¥–æ—Å—Ç—É–ø–µ–Ω –∫–∞–∫ –æ–±—ã—á–Ω—ã–π —Ç—Ä–∞–Ω—Å—Ñ–æ—Ä–º–µ—Ä

def encode_texts(texts):
    enc = tokenizer(
        texts,
        padding=True,
        truncation=True,
        return_tensors="pt",
    )
    with torch.no_grad():
        outputs = base_model(**enc)
    # outputs.last_hidden_state: [batch, seq_len, hidden_size]
    token_embeddings = outputs.last_hidden_state
    attention_mask = enc["attention_mask"]
    # mean pooling
    input_mask_expanded = attention_mask.unsqueeze(-1).expand(token_embeddings.size()).float()
    sum_embeddings = torch.sum(token_embeddings * input_mask_expanded, dim=1)
    sum_mask = torch.clamp(input_mask_expanded.sum(dim=1), min=1e-9)
    embeddings = sum_embeddings / sum_mask
    return embeddings

emb = encode_texts(texts)
print(emb.shape)  # [3, hidden_size]

def split_into_chunks(doc, max_chars=500):
    # –æ—á–µ–Ω—å –ø—Ä–æ—Å—Ç–æ–π –≤–∞—Ä–∏–∞–Ω—Ç: —Ä–µ–∂–µ–º –ø–æ '\n\n'

    chunks = []
    for i, p in enumerate(paragraphs):
        if len(p) <= max_chars:
            chunks.append({
                "doc_id": doc["id"],
                "chunk_id": i,
                "text": p,
            })
        else:
            # –º–æ–∂–Ω–æ –µ—â–µ —Ä–∞–∑–±–∏—Ç—å –¥–ª–∏–Ω–Ω—ã–π –ø–∞—Ä–∞–≥—Ä–∞—Ñ –Ω–∞ –∫—É—Å–∫–∏ –ø–æ —Å–∏–º–≤–æ–ª–∞–º

                chunks.append({
                    "doc_id": doc["id"],
                    "chunk_id": f"{i}_{j}",
                    "text": p[j:j+max_chars],
                })
    return chunks

def search_kb(query, kb_index, top_k=5):
    q_emb = embed_model.encode([query], normalize_embeddings=True)[0]  # (d,)
    kb_emb = kb_index["embeddings"]  # (N, d)

    scores = np.dot(kb_emb, q_emb)  # –ø—Ä–∏ –Ω–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –≤–µ–∫—Ç–æ—Ä–∞—Ö —ç—Ç–æ cosine

    results = []
    for idx in top_idx:
        chunk = kb_index["chunks"][idx]
        score = scores[idx]
        results.append({
            "score": float(score),
            "doc_id": chunk["doc_id"],
            "chunk_id": chunk["chunk_id"],
            "text": chunk["text"],
        })
    return results

        text,
        return_tensors="pt",
        truncation=True,

def search_kb(query, kb_index, top_k=5):
    q_emb = embed_model.encode([query], normalize_embeddings=True)[0]
    kb_emb = kb_index["embeddings"]
    scores = np.dot(kb_emb, q_emb)
    top_idx = np.argsort(scores)[::-1][:top_k]
    results = []
    for idx in top_idx:
        chunk = kb_index["chunks"][idx]
        score = scores[idx]
        results.append({
            "score": float(score),
            "doc_id": chunk["doc_id"],
            "chunk_id": chunk["chunk_id"],
            "text": chunk["text"],
        })
    return results

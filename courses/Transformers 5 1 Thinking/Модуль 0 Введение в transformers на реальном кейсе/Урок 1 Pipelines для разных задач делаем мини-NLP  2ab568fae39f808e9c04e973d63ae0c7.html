from transformers import pipeline

# –ó–∞–≥—Ä—É–∂–∞–µ–º –≤—Å–µ –Ω—É–∂–Ω—ã–µ –ø–∞–π–ø–ª–∞–π–Ω—ã –æ–¥–∏–Ω —Ä–∞–∑
ner = pipeline("ner", grouped_entities=True)
qa = pipeline("question-answering")
summarizer = pipeline("summarization")
translator_en_ru = pipeline(
    "translation_en_to_ru",
    model="Helsinki-NLP/opus-mt-en-ru"
)

def extract_entities(text: str):
    return ner(text)

def answer_question(question: str, context: str):
    return qa(question=question, context=context)

def summarize_text(text: str, max_length=60, min_length=20):
    return summarizer(text, max_length=max_length, min_length=min_length, do_sample=False)[0]["summary_text"]

def translate_en_to_ru(text: str):
    return translator_en_ru(text, max_length=100)[0]["translation_text"]

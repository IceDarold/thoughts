import torch
from transformers import AutoTokenizer, AutoModelForSequenceClassification
import torch.nn.functional as F

model_path = "./sentiment-imdb-distilbert/best"

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

tokenizer = AutoTokenizer.from_pretrained(model_path)
model = AutoModelForSequenceClassification.from_pretrained(model_path).to(device)
model.eval()

def predict_sentiment(texts):
    if isinstance(texts, str):
        texts = [texts]

    batch = tokenizer(
        texts,
        padding=True,
        truncation=True,
        max_length=256,
        return_tensors="pt"
    ).to(device)

    with torch.no_grad():
        outputs = model(**batch)

    logits = outputs.logits
    probs = F.softmax(logits, dim=-1)
    pred_ids = torch.argmax(probs, dim=-1)

    results = []
    id2label = model.config.id2label

    for text, pred_id, prob_vec in zip(texts, pred_ids, probs):
        label = id2label[pred_id.item()]
        confidence = prob_vec[pred_id].item()
        results.append({
            "text": text,
            "label": label,
            "score": confidence,
        })

    return results


if __name__ == "__main__":
    examples = [
        "I absolutely loved this movie, it was amazing!",
        "This was a total waste of time.",
        "It was fine, but I wouldn't watch it again."
    ]

    preds = predict_sentiment(examples)
    for p in preds:
        print(p)

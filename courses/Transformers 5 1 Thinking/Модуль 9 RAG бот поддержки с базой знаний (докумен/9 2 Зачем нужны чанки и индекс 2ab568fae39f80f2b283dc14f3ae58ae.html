def split_into_chunks(doc, max_chars=500):
    """
    doc: {"id": ..., "text": ...}
    max_chars: –º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –¥–ª–∏–Ω–∞ —á–∞–Ω–∫–∞ –≤ —Å–∏–º–≤–æ–ª–∞—Ö
    """
    paragraphs = [p.strip() for p in doc["text"].split("\n\n") if p.strip()]
    chunks = []

    for i, p in enumerate(paragraphs):
        if len(p) <= max_chars:
            chunks.append({
                "doc_id": doc["id"],
                "chunk_id": i,
                "text": p,
            })
        else:
            # –î–ª–∏–Ω–Ω—ã–π –ø–∞—Ä–∞–≥—Ä–∞—Ñ —Ä–µ–∂–µ–º –ø–æ max_chars
            for j in range(0, len(p), max_chars):
                part = p[j : j + max_chars]
                chunks.append({
                    "doc_id": doc["id"],
                    "chunk_id": f"{i}_{j}",  # —Å–ª–æ–∂–Ω—ã–π id, –Ω–æ –Ω–∞–º –Ω–æ—Ä–º
                    "text": part,
                })

    return chunks

# –ø—Ä–∏–º–µ–Ω—è–µ–º –∫ –≤—Å–µ–º –¥–æ–∫—É–º–µ–Ω—Ç–∞–º

    # 2. –ö–æ—Å–∏–Ω—É—Å–Ω–∞—è –ø–æ—Ö–æ–∂–µ—Å—Ç—å: –ø—Ä–∏ –Ω–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –≤–µ–∫—Ç–æ—Ä–∞—Ö —ç—Ç–æ –ø—Ä–æ—Å—Ç–æ dot product
    scores = np.dot(kb_emb, q_emb)  # (N,)

    # 3. –ò–Ω–¥–µ–∫—Å—ã —Ç–æ–ø-k
    top_idx = np.argsort(scores)[::-1][:top_k]  # —Å–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ —É–±—ã–≤–∞–Ω–∏—é

    results = []
    for idx in top_idx:
        chunk = kb_index["chunks"][idx]
        score = scores[idx]
        results.append({
            "score": float(score),
            "doc_id": chunk["doc_id"],
            "chunk_id": chunk["chunk_id"],
            "text": chunk["text"],
        })

    return results

import faiss

# –ü—É—Å—Ç—å embeddings —É–∂–µ –≤ float32
kb_emb_f32 = embeddings.astype("float32")
dim = kb_emb_f32.shape[1]

# IndexFlatIP ‚Äî –ø—Ä–æ—Å—Ç–µ–π—à–∏–π –∏–Ω–¥–µ–∫—Å –ø–æ inner product (–ø–æ–¥—Ö–æ–¥–∏—Ç –¥–ª—è normalized vectors)

def search_kb_faiss(query, kb_index, embed_model, index, top_k=5):
    q_emb = embed_model.encode([query], normalize_embeddings=True).astype("float32")  # (1, dim)
    scores, idxs = index.search(q_emb, top_k)  # scores: (1, top_k), idxs: (1, top_k)

    scores = scores[0]
    idxs = idxs[0]

    results = []
    for score, idx in zip(scores, idxs):
        chunk = kb_index["chunks"][idx]
        results.append({
            "score": float(score),
            "doc_id": chunk["doc_id"],
            "chunk_id": chunk["chunk_id"],
            "text": chunk["text"],
        })
    return results

def build_context_for_query(query, kb_index, embed_model, top_k=3):
    """
    –ò—â–µ—Ç —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–µ —á–∞–Ω–∫–∏ –≤ –±–∞–∑–µ –∑–Ω–∞–Ω–∏–π –∏ —Å–æ–±–∏—Ä–∞–µ—Ç –∏—Ö –≤ –æ–¥–∏–Ω —Ç–µ–∫—Å—Ç–æ–≤—ã–π –±–ª–æ–∫.

    """

def build_rag_prompt(question, context):
    """
    –°—Ç—Ä–æ–∏—Ç –ø—Ä–æ–º–ø—Ç –¥–ª—è RAG: system-–∏–Ω—Å—Ç—Ä—É–∫—Ü–∏—è + –∫–æ–Ω—Ç–µ–∫—Å—Ç –∏–∑ KB + –≤–æ–ø—Ä–æ—Å.
    """
    system_instruction = (
        "System: You are a helpful support assistant for a mobile app. "
        "Use only the information from the context below to answer the question. "
        "If the answer is not in the context, say that you don't know."
    )

    prompt = (
        system_instruction
        + "\n\nContext:\n"
        + context
        + "\n\nQuestion: "
        + question.strip()
        + "\nAnswer:"
    )

    return prompt

import torch

def generate_answer_from_prompt(
    prompt,
    bot_model,
    bot_tokenizer,
    max_new_tokens=128,
    temperature=0.7,
    top_p=0.9,
):
    inputs = bot_tokenizer(
        prompt,
        return_tensors="pt",
        truncation=True,
        max_length=1024,
    ).to(bot_model.device)

    with torch.no_grad():
        out_ids = bot_model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            do_sample=True,
            top_p=top_p,
            temperature=temperature,
            repetition_penalty=1.1,
            eos_token_id=bot_tokenizer.eos_token_id,
        )

    full_text = bot_tokenizer.decode(out_ids[0], skip_special_tokens=True)

    # –û–±—Ä–µ–∑–∞–µ–º –≤—Å—ë, —á—Ç–æ –±—ã–ª–æ –¥–æ "Answer:" (–µ—Å–ª–∏ –º–æ–¥–µ–ª—å –µ–≥–æ –ø–æ–≤—Ç–æ—Ä–∏—Ç)
    if "Answer:" in full_text:
        answer = full_text.split("Answer:", 1)[-1].strip()
    else:
        # fallback: –ø—Ä–æ—Å—Ç–æ –±–µ—Ä—ë–º —Ö–≤–æ—Å—Ç –ø–æ—Å–ª–µ –∏—Å—Ö–æ–¥–Ω–æ–≥–æ –ø—Ä–æ–º–ø—Ç–∞
        answer = full_text[len(prompt):].strip()

    return answer, full_text

        "context": context,

        "max_score": max_score,
    }

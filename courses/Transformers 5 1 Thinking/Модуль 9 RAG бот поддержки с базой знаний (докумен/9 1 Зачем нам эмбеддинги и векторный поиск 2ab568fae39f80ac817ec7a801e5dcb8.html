for i, j in pairs:

from transformers import AutoTokenizer, AutoModel

# –ú–æ–∂–Ω–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å —Ç—É –∂–µ –º–æ–¥–µ–ª—å, —á—Ç–æ –∏ –≤ sentence-transformers

base_model = AutoModel.from_pretrained(model_id)

texts = [
    "The app crashes when I open the gallery.",
    "Application error when viewing photos.",
    "How can I change my subscription plan?",
    "What is the refund policy for annual subscriptions?",
]

def encode_texts(texts):
    # 1. –¢–æ–∫–µ–Ω–∏–∑–∞—Ü–∏—è
    enc = tokenizer(
        texts,
        padding=True,
        truncation=True,
        return_tensors="pt",

    # 2. –≠–º–±–µ–¥–¥–∏–Ω–≥–∏ —Ç–æ–∫–µ–Ω–æ–≤: [batch, seq_len, hidden_size]

    # 3. Attention mask: –≥–¥–µ 1 ‚Äî —Ä–µ–∞–ª—å–Ω—ã–π —Ç–æ–∫–µ–Ω, –≥–¥–µ 0 ‚Äî –ø–∞–¥–¥–∏–Ω–≥

    # 4. Mean pooling: —É—Å—Ä–µ–¥–Ω—è–µ–º —Ç–æ–ª—å–∫–æ –ø–æ —Ä–µ–∞–ª—å–Ω—ã–º —Ç–æ–∫–µ–Ω–∞–º

    return embeddings

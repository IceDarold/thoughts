def classify_ticket(text: str):
    """
    –ö–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ—Ç —Ç–∏–∫–µ—Ç/–¥–∏–∞–ª–æ–≥ –ø–æ –∫–ª–∞—Å—Å–∞–º:
    bug / feature / billing / other (–∫–∞–∫ —É —Ç–µ–±—è –æ–±—É—á–µ–Ω–æ).
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç dict —Å label –∏ confidence.
    """
    enc = cls_tokenizer(
        text,
        return_tensors="pt",
        truncation=True,
        max_length=256,
    ).to(cls_model.device)

    with torch.no_grad():
        outputs = cls_model(**enc)

    logits = outputs.logits
    probs = torch.softmax(logits, dim=-1)[0]
    pred_id = torch.argmax(probs).item()
    id2label = cls_model.config.id2label

    return {
        "label": id2label[pred_id],
        "confidence": float(probs[pred_id]),
    }

def summarize_dialog(dialog: str, max_new_tokens: int = 64) -> str:
    """
    –î–µ–ª–∞–µ—Ç –∫—Ä–∞—Ç–∫–æ–µ summary –¥–∏–∞–ª–æ–≥–∞/—Ç–∏–∫–µ—Ç–∞.

    inputs = summ_tokenizer(
        dialog,
        return_tensors="pt",
        truncation=True,
        max_length=512,
    ).to(device)

    with torch.no_grad():
        out_ids = summ_model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            num_beams=4,
            no_repeat_ngram_size=3,
        )

    summary = summ_tokenizer.decode(out_ids[0], skip_special_tokens=True)
    return summary

def search_kb(query, kb_index, embed_model, top_k=5):
    """
    –ò—â–µ—Ç —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–µ —á–∞–Ω–∫–∏ –≤ KB –ø–æ —Ç–µ–∫—Å—Ç–æ–≤–æ–º—É –∑–∞–ø—Ä–æ—Å—É.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ø–∏—Å–æ–∫ dict'–æ–≤: score, doc_id, chunk_id, text.
    """
    q_emb = embed_model.encode([query], normalize_embeddings=True)[0]
    kb_emb = kb_index["embeddings"]  # (N, dim)

    scores = np.dot(kb_emb, q_emb)  # –ø—Ä–∏ normalized: cosine similarity
    top_idx = np.argsort(scores)[::-1][:top_k]

    results = []
    for idx in top_idx:
        chunk = kb_index["chunks"][idx]
        score = scores[idx]
        results.append({
            "score": float(score),
            "doc_id": chunk["doc_id"],
            "chunk_id": chunk["chunk_id"],
            "text": chunk["text"],
        })
    return results

def build_context(question: str, ticket_class: str, summary: str, kb_index, embed_model, top_k=3):
    """
    –°—Ç—Ä–æ–∏—Ç –ø–æ–∏—Å–∫–æ–≤—ã–π –∑–∞–ø—Ä–æ—Å –∏ –¥–æ—Å—Ç–∞–µ—Ç top-k —á–∞–Ω–∫–æ–≤ –∏–∑ KB.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
      - context (str) ‚Äî –≥–æ—Ç–æ–≤—ã–π —Ç–µ–∫—Å—Ç –¥–ª—è –ø—Ä–æ–º–ø—Ç–∞
      - results (list) ‚Äî –ø–æ–¥—Ä–æ–±–Ω–æ—Å—Ç–∏ —á–∞–Ω–∫–æ–≤.
    """

    # –ú–æ–∂–Ω–æ –ø–æ–∏–≥—Ä–∞—Ç—å—Å—è —Å —Ç–µ–º, –∫–∞–∫ —Ñ–æ—Ä–º–∏—Ä–æ–≤–∞—Ç—å query.
    # –ù–∞–ø—Ä–∏–º–µ—Ä: "[BUG] summary" + "full question"
    search_query = f"[{ticket_class}] {summary if summary else ''} {question}"

    results = search_kb(search_query, kb_index, embed_model, top_k=top_k)

    blocks = []
    for r in results:
        header = f"[{r['doc_id']}#{r['chunk_id']} | score={r['score']:.3f}]"
        blocks.append(header + "\n" + r["text"])

    context = "\n\n---\n\n".join(blocks)
    return context, results

def generate_support_reply(prompt: str, max_new_tokens: int = 128) -> str:
    inputs = bot_tokenizer(
        prompt,
        return_tensors="pt",
        truncation=True,
        max_length=1024,
    ).to(bot_model.device)

    with torch.no_grad():
        out_ids = bot_model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            do_sample=True,
            top_p=0.9,
            temperature=0.7,
            repetition_penalty=1.1,
            eos_token_id=bot_tokenizer.eos_token_id,
        )

    full_text = bot_tokenizer.decode(out_ids[0], skip_special_tokens=True)

    if "Support answer:" in full_text:
        reply = full_text.split("Support answer:", 1)[-1].strip()
    else:
        reply = full_text[len(prompt):].strip()

    return reply, full_text

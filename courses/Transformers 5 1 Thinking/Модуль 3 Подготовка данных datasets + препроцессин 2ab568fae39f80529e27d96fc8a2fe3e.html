3.4.2. –ö–∞—Ä–∫–∞—Å data_pipeline.py

from typing import Dict, Tuple
from datasets import load_dataset, DatasetDict
from transformers import AutoTokenizer, DataCollatorWithPadding


def load_and_prepare_data(
    csv_path: str,
    model_name: str = "distilbert-base-uncased",
    text_column: str = "text",
    label_column: str = "label",
    test_size: float = 0.1,
    max_length: int = 128,
    seed: int = 42,
) -> Tuple[DatasetDict, Dict[str, int], Dict[int, str], AutoTokenizer, DataCollatorWithPadding]:
    # 1. –ó–∞–≥—Ä—É–∂–∞–µ–º CSV
    raw = load_dataset("csv", data_files=csv_path)["train"]

    # 2. –†–∞–∑–±–∏–≤–∞–µ–º –Ω–∞ train/validation
    split = raw.train_test_split(test_size=test_size, seed=seed)
    dataset = DatasetDict(
        {
            "train": split["train"],
            "validation": split["test"],
        }
    )

    # 3. –°—Ç—Ä–æ–∏–º label_list, label2id, id2label
    #    (–µ—Å–ª–∏ –º–µ—Ç–∫–∏ —Å—Ç—Ä–æ–∫–æ–≤—ã–µ)
    unique_labels = sorted(set(dataset["train"][label_column]))
    label2id = {l: i for i, l in enumerate(unique_labels)}
    id2label = {i: l for l, i in label2id.items()}

    # 4. –¢–æ–∫–µ–Ω–∏–∑–∞—Ç–æ—Ä
    tokenizer = AutoTokenizer.from_pretrained(model_name)

    # 5. –§—É–Ω–∫—Ü–∏—è –ø—Ä–µ–ø—Ä–æ—Ü–µ—Å—Å–∏–Ω–≥–∞
    def preprocess_function(batch):
        texts = batch[text_column]
        tokenized = tokenizer(
            texts,
            truncation=True,
            # –ü–∞–¥–¥–∏–Ω–≥ –ª—É—á—à–µ –¥–µ–ª–µ–≥–∏—Ä–æ–≤–∞—Ç—å collator'—É:
            padding=False,
            max_length=max_length,
        )
        tokenized["labels"] = [label2id[l] for l in batch[label_column]]
        return tokenized

    # 6. –ü—Ä–∏–º–µ–Ω—è–µ–º map
    tokenized_dataset = dataset.map(
        preprocess_function,
        batched=True,
        remove_columns=[text_column, label_column],
    )

    # 7. DataCollator
    data_collator = DataCollatorWithPadding(tokenizer=tokenizer, return_tensors="pt")

    return tokenized_dataset, label2id, id2label, tokenizer, data_collator

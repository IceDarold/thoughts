def messages_to_text(messages):
    """
    –ü—Ä–µ–≤—Ä–∞—â–∞–µ–º —Å–ø–∏—Å–æ–∫ —Å–æ–æ–±—â–µ–Ω–∏–π —Ñ–æ—Ä–º–∞—Ç–∞:
    [{"role": "user", "content": "..."} ...]
    –≤ –æ–¥–∏–Ω —Ç–µ–∫—Å—Ç–æ–≤—ã–π –¥–∏–∞–ª–æ–≥.
    """
    lines = []
    for msg in messages:
        role = msg["role"]
        content = msg["content"].strip()

        if role == "system":
            lines.append(f"System: {content}")
        elif role == "user":
            lines.append(f"User: {content}")
        else:  # assistant / support
            lines.append(f"Support: {content}")

    # –í –∫–æ–Ω—Ü–µ ‚Äî –ø–µ—Ä–µ–Ω–æ—Å —Å—Ç—Ä–æ–∫–∏, —á—Ç–æ–±—ã –º–æ–¥–µ–ª—å –≤–∏–¥–µ–ª–∞ —Å—Ç—Ä—É–∫—Ç—É—Ä—É
    return "\n".join(lines)

import torch

def build_prompt(messages):
    """
    messages: —Å–ø–∏—Å–æ–∫ {"role": "...", "content": "..."}
    """
    lines = []
    for msg in messages:
        role = msg["role"]
        content = msg["content"].strip()
        if role == "system":
            lines.append(f"System: {content}")
        elif role == "user":
            lines.append(f"User: {content}")
        else:
            lines.append(f"Support: {content}")
    # –¥–æ–±–∞–≤–ª—è–µ–º –º–∞—Ä–∫–µ—Ä –Ω–∞—á–∞–ª–∞ –Ω–æ–≤–æ–π —Ä–µ–ø–ª–∏–∫–∏ —Å–∞–ø–ø–æ—Ä—Ç–∞
    prompt = "\n".join(lines) + "\nSupport:"
    return prompt


def generate_support_reply(peft_model, tokenizer, messages, max_new_tokens=80):
    device = next(peft_model.parameters()).device

    prompt = build_prompt(messages)
    inputs = tokenizer(prompt, return_tensors="pt", truncation=True, max_length=512).to(device)

    with torch.no_grad():
        output_ids = peft_model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            do_sample=True,
            top_p=0.9,
            temperature=0.7,
            repetition_penalty=1.1,
            eos_token_id=tokenizer.eos_token_id,
        )

    # –û–±—Ä–µ–∑–∞–µ–º —Ç–æ–ª—å–∫–æ "–Ω–æ–≤—É—é" —á–∞—Å—Ç—å
    input_len = inputs["input_ids"].shape[1]
    new_tokens = output_ids[0][input_len:]
    reply_text = tokenizer.decode(new_tokens, skip_special_tokens=True).strip()

    # –ù–∞ —Å–ª—É—á–∞–π, –µ—Å–ª–∏ –º–æ–¥–µ–ª—å –Ω–∞—á–∞–ª–∞ –ø–∏—Å–∞—Ç—å "User:" ‚Äî –æ–±—Ä–µ–∂–µ–º –ø–æ —ç—Ç–æ–º—É –º–∞—Ä–∫–µ—Ä—É
    if "User:" in reply_text:
        reply_text = reply_text.split("User:")[0].strip()

    return reply_text

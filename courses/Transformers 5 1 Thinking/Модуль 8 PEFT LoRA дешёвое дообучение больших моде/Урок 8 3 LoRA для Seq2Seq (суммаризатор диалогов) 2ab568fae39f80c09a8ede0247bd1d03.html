    inputs = batch["dialog"]
    targets = batch["summary"]

    # –¢–æ–∫–µ–Ω–∏–∑–∞—Ü–∏—è –≤—Ö–æ–¥–∞ (–¥–∏–∞–ª–æ–≥–∞)

def compute_metrics(eval_pred):
    predictions, labels = eval_pred

    # –ï—Å–ª–∏ –º–æ–¥–µ–ª—å –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç Tuple, –±–µ—Ä—ë–º –ø–µ—Ä–≤—É—é —á–∞—Å—Ç—å
    if isinstance(predictions, tuple):
        predictions = predictions[0]

    # –î–µ–∫–æ–¥–∏—Ä—É–µ–º –ø—Ä–µ–¥–∏–∫—Ç—ã
    decoded_preds = tokenizer.batch_decode(
        predictions, skip_special_tokens=True
    )

    # –î–µ–∫–æ–¥–∏—Ä—É–µ–º labels: –∑–∞–º–µ–Ω—è–µ–º -100 –Ω–∞ pad_token_id
    labels = np.where(labels != -100, labels, tokenizer.pad_token_id)
    decoded_labels = tokenizer.batch_decode(
        labels, skip_special_tokens=True
    )

    decoded_preds, decoded_labels = postprocess_text(decoded_preds, decoded_labels)

    result = rouge_metric.compute(
        predictions=decoded_preds,
        references=decoded_labels,
        use_stemmer=True,
    )

    # –û–±—ã—á–Ω–æ —Å–º–æ—Ç—Ä—è—Ç rouge1/rouge2/rougeL
    result = {k: v.mid.fmeasure for k, v in result.items()}

    return result

def summarize_dialog_lora(dialog: str, max_new_tokens: int = 64) -> str:
    inputs = tokenizer(
        dialog,
        return_tensors="pt",
        truncation=True,
        max_length=512,
    ).to(device)

    with torch.no_grad():
        out_ids = lora_model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            num_beams=4,
            no_repeat_ngram_size=3,
        )

    summary = tokenizer.decode(
        out_ids[0],
        skip_special_tokens=True,
    ).strip()

    return summary

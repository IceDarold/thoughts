def classify_batch(texts: list[str]):
    """
    –ë–∞—Ç—á-–∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è —Å–ø–∏—Å–∫–∞ —Ç–µ–∫—Å—Ç–æ–≤.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ø–∏—Å–æ–∫ —Å–ª–æ–≤–∞—Ä–µ–π:
        {"text": ..., "label": ..., "confidence": ...}
    """
    if len(texts) == 0:
        return []

    # 1. –¢–æ–∫–µ–Ω–∏–∑–∏—Ä—É–µ–º –≤—Å—ë —Å—Ä–∞–∑—É
    enc = cls_tokenizer(
        texts,
        return_tensors="pt",
        truncation=True,
        padding=True,      # –≤–∞–∂–Ω—ã–π –º–æ–º–µ–Ω—Ç: –±–∞—Ç—á ‚Üí –Ω—É–∂–µ–Ω padding
        max_length=256,
    ).to(cls_model.device)

    # 2. –ò–Ω—Ñ–µ—Ä–µ–Ω—Å –±–µ–∑ –≥—Ä–∞–¥–∏–µ–Ω—Ç–æ–≤
    with torch.no_grad():
        outputs = cls_model(**enc)

    logits = outputs.logits             # shape: [batch_size, num_labels]
    probs = torch.softmax(logits, dim=-1)

    id2label = cls_model.config.id2label

    results = []
    for i, text in enumerate(texts):
        p = probs[i]                    # –≤–µ–∫—Ç–æ—Ä –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π –¥–ª—è i-–≥–æ —Ç–µ–∫—Å—Ç–∞
        pred_id = torch.argmax(p).item()
        results.append({
            "text": text,
            "label": id2label[pred_id],
            "confidence": float(p[pred_id]),
        })
    return results

import time

from app import classify_ticket, classify_batch


def generate_fake_tickets(n: int):
    base_text = "After the last update the app crashes when I open the gallery."
    # –î–ª—è –ø—Ä–æ—Å—Ç–æ—Ç—ã –¥–µ–ª–∞–µ–º –æ–¥–∏–Ω–∞–∫–æ–≤—ã–µ, –Ω–∞–º –≤–∞–∂–Ω–∞ —Å–∫–æ—Ä–æ—Å—Ç—å, –∞ –Ω–µ —Å–º—ã—Å–ª
    return [base_text for _ in range(n)]


def benchmark_single(tickets):
    start = time.time()
    results = []
    for t in tickets:
        label, conf = classify_ticket(t)
        results.append((label, conf))
    total_time = time.time() - start
    return total_time, results


def benchmark_batch(tickets):
    start = time.time()
    results = classify_batch(tickets)
    total_time = time.time() - start
    return total_time, results

import asyncio
from typing import List, Any
from dataclasses import dataclass

from app import classify_batch  # –Ω–∞—à–∞ –±–∞—Ç—á-—Ñ—É–Ω–∫—Ü–∏—è


@dataclass
class QueueItem:
    text: str
    future: asyncio.Future  # —Å—é–¥–∞ –ø–æ–ª–æ–∂–∏–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç {"label": ..., "confidence": ...}


class BatchProcessor:
    def __init__(self, batch_size: int = 16, max_wait_ms: int = 50):
        self.queue: asyncio.Queue[QueueItem] = asyncio.Queue()
        self.batch_size = batch_size
        self.max_wait_ms = max_wait_ms

    async def add_request(self, text: str) -> Any:
        loop = asyncio.get_event_loop()
        fut = loop.create_future()
        item = QueueItem(text=text, future=fut)
        await self.queue.put(item)
        # –∂–¥—ë–º, –ø–æ–∫–∞ –≤–æ—Ä–∫–µ—Ä –ø–æ–ª–æ–∂–∏—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç –≤ future
        result = await fut
        return result

    async def worker_loop(self):
        while True:
            batch_items: List[QueueItem] = []

            # 1. –ë–ª–æ–∫–∏—Ä—É—é—â–µ –∂–¥—ë–º –ø–µ—Ä–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç
            first_item = await self.queue.get()
            batch_items.append(first_item)

            # 2. –ü—ã—Ç–∞–µ–º—Å—è —Å–æ–±—Ä–∞—Ç—å –µ—â—ë —ç–ª–µ–º–µ–Ω—Ç—ã –¥–æ batch_size –∏–ª–∏ –¥–æ —Ç–∞–π–º–∞—É—Ç–∞
            start = asyncio.get_event_loop().time()
            while len(batch_items) < self.batch_size:
                timeout = self.max_wait_ms / 1000.0
                elapsed = asyncio.get_event_loop().time() - start
                remaining = timeout - elapsed
                if remaining <= 0:
                    break
                try:
                    item = await asyncio.wait_for(self.queue.get(), timeout=remaining)
                    batch_items.append(item)
                except asyncio.TimeoutError:
                    break

            # 3. –£ –Ω–∞—Å –µ—Å—Ç—å batch_items ‚Äî –¥–µ–ª–∞–µ–º –±–∞—Ç—á-–∏–Ω—Ñ–µ—Ä–µ–Ω—Å
            texts = [it.text for it in batch_items]
            batch_results = classify_batch(texts)  # —Å–∏–Ω—Ö—Ä–æ–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è, –¥–ª—è –ø—Ä–æ—Å—Ç–æ—Ç—ã

            # 4. –†–∞—Å–∫–∏–¥—ã–≤–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø–æ future
            for item, res in zip(batch_items, batch_results):
                if not item.future.done():
                    item.future.set_result(res)

    """
    –õ–æ–≥–∏—Ä—É–µ–º –æ–¥–∏–Ω –∑–∞–ø—Ä–æ—Å/–æ—Ç–≤–µ—Ç –≤ —Ñ–∞–π–ª –≤–∏–¥–∞ logs/{endpoint}.log

# eval_classifier.py
import json
from pathlib import Path

from sklearn.metrics import accuracy_score, f1_score, classification_report, confusion_matrix

from app import classify_ticket  # –∏–º–ø–æ—Ä—Ç–∏—Ä—É–µ–º —Ñ—É–Ω–∫—Ü–∏—é –∏ –º–æ–¥–µ–ª–∏


def load_eval_data(path: str):
    data = []
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            if not line.strip():
                continue
            item = json.loads(line)
            data.append(item)
    return data


def evaluate_classifier(eval_path: str = "eval_tickets.jsonl"):
    data = load_eval_data(eval_path)

    y_true = []
    y_pred = []

    for item in data:
        dialog = item["dialog"]
        true_label = item["expected_label"]
        pred_label, conf = classify_ticket(dialog)

        y_true.append(true_label)
        y_pred.append(pred_label)

    acc = accuracy_score(y_true, y_pred)
    f1 = f1_score(y_true, y_pred, average="weighted")

    print(f"Num samples: {len(data)}")
    print(f"Accuracy: {acc:.4f}")
    print(f"F1 (weighted): {f1:.4f}")
    print("\nClassification report:")
    print(classification_report(y_true, y_pred))
    print("\nConfusion matrix:")
    print(confusion_matrix(y_true, y_pred))


if __name__ == "__main__":
    evaluate_classifier()

import json

import numpy as np
import evaluate  # pip install evaluate

from app import summarize_dialog  # –∏—Å–ø–æ–ª—å–∑—É–µ–º –º–æ–¥–µ–ª—å —Å—É–º–º–∞—Ä–∏–∑–∞—Ü–∏–∏


def load_eval_summaries(path: str = "eval_summaries.jsonl"):
    data = []
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            if not line.strip():
                continue
            item = json.loads(line)
            data.append(item)
    return data


def evaluate_summarizer(eval_path: str = "eval_summaries.jsonl"):
    data = load_eval_summaries(eval_path)

    preds = []
    refs = []

    for item in data:
        dialog = item["dialog"]
        ref_summary = item["reference_summary"]

        pred_summary = summarize_dialog(dialog, max_new_tokens=64)

        preds.append(pred_summary.strip())
        refs.append(ref_summary.strip())

    rouge = evaluate.load("rouge")
    result = rouge.compute(predictions=preds, references=refs, use_stemmer=True)

    # result: dict with rouge1/rouge2/rougeL/rougeLsum
    print(f"Num samples: {len(data)}")
    for k, v in result.items():
        # v - –æ–±—ä–µ–∫—Ç —Å precision/recall/fmeasure, –Ω–æ evaluate –¥–∞—ë—Ç —Å—Ä–∞–∑—É —á–∏—Å–ª–∞ (–∏–ª–∏ mid)
        print(f"{k}: {v:.4f}")


if __name__ == "__main__":
    evaluate_summarizer()

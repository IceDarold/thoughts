def merge_wordpieces_and_labels(tokens, pred_ids, offsets, text, id2label):
    entities = []
    current_entity = None

    for token, pred_id, (start, end) in zip(tokens, pred_ids, offsets):
        start = start.item()
        end = end.item()
        label = id2label[pred_id.item()]

        if end == 0 and start == 0:
            # —Å–ø–µ—Ü-—Ç–æ–∫–µ–Ω ([CLS], [SEP], –ø–∞–¥–¥–∏–Ω–≥)
            continue

        # BIO —Å—Ö–µ–º–∞: label –º–æ–∂–µ—Ç –±—ã—Ç—å 'B-ORG', 'I-ORG', 'O'
        if label == "O":
            # –∑–∞–∫—Ä—ã–≤–∞–µ–º —Ç–µ–∫—É—â—É—é —Å—É—â–Ω–æ—Å—Ç—å, –µ—Å–ª–∏ –±—ã–ª–∞
            if current_entity is not None:
                entities.append(current_entity)
                current_entity = None
            continue

        # –ò–Ω–∞—á–µ —Å—É—â–Ω–æ—Å—Ç—å:
        label_prefix, label_type = label.split("-", maxsplit=1)

        if label_prefix == "B" or current_entity is None or current_entity["type"] != label_type:
            # –Ω–∞—á–∏–Ω–∞–µ–º –Ω–æ–≤—É—é —Å—É—â–Ω–æ—Å—Ç—å
            if current_entity is not None:
                entities.append(current_entity)
            current_entity = {
                "type": label_type,
                "start": start,
                "end": end,
            }
        else:
            # –ø—Ä–æ–¥–æ–ª–∂–∞–µ–º —Å—É—â–Ω–æ—Å—Ç—å (I-...)
            current_entity["end"] = end

    # –Ω–µ –∑–∞–±—ã–≤–∞–µ–º –¥–æ–±–∞–≤–∏—Ç—å –ø–æ—Å–ª–µ–¥–Ω—é—é
    if current_entity is not None:
        entities.append(current_entity)

    # –¥–æ–±–∞–≤–∏–º —Ç–µ–∫—Å—Ç —Å—É—â–Ω–æ—Å—Ç–∏
    for ent in entities:
        ent["text"] = text[ent["start"]:ent["end"]]

    return entities

def extract_ner_entities(text: str, tokenizer, model):
    enc = tokenizer(
        text,
        return_tensors="pt",
        padding=True,
        truncation=True,
        return_offsets_mapping=True,
    )

    offsets = enc.pop("offset_mapping")[0]  # [seq_len, 2]

    with torch.no_grad():
        outputs = model(**enc)

    logits = outputs.logits  # [1, seq_len, num_labels]
    pred_ids = torch.argmax(logits, dim=-1)[0]  # [seq_len]

    tokens = tokenizer.convert_ids_to_tokens(enc["input_ids"][0])
    id2label = model.config.id2label

    entities = merge_wordpieces_and_labels(tokens, pred_ids, offsets, text, id2label)
    return entities

def answer_from_faq(question: str, faqs: list[dict], qa_pipe, min_score=0.1):
    candidates = []
    for faq in faqs:
        res = qa_pipe(question=question, context=faq["text"])
        candidates.append({
            "title": faq["title"],
            "answer": res["answer"],
            "score": res["score"],
        })

    # —Å–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ score
    candidates = sorted(candidates, key=lambda x: x["score"], reverse=True)
    best = candidates[0]

    if best["score"] < min_score:
        return {"answer": None, "source": None, "score": best["score"]}

    return {
        "answer": best["answer"],
        "source": best["title"],
        "score": best["score"],
    }

q1 = "Why was I charged twice this month?"
q2 = "How can I reset my password?"

print(answer_from_faq(q1, faqs, qa))
print(answer_from_faq(q2, faqs, qa))

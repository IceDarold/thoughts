def train_with_accelerate(
    csv_path: str,
    model_name: str = "distilbert-base-uncased",
    num_epochs: int = 3,
    lr: float = 2e-5,
    batch_size: int = 16,
):
    # 1. –°–æ–∑–¥–∞—ë–º Accelerator
    accelerator = Accelerator()
    accelerator.print("Using device:", accelerator.device)

    # 2. –ó–∞–≥—Ä—É–∂–∞–µ–º –¥–∞–Ω–Ω—ã–µ
    tokenized_dataset, label2id, id2label, tokenizer, data_collator = load_and_prepare_data(
        csv_path=csv_path,
        model_name=model_name,
        max_length=128,
        test_size=0.1,
        seed=42,
    )

    train_dataset = tokenized_dataset["train"]
    val_dataset = tokenized_dataset["validation"]

    # 3. Dataloader'—ã
    train_dataloader = DataLoader(
        train_dataset,
        batch_size=batch_size,
        shuffle=True,
        collate_fn=data_collator,
    )

    val_dataloader = DataLoader(
        val_dataset,
        batch_size=batch_size,
        shuffle=False,
        collate_fn=data_collator,
    )

    # 4. –ú–æ–¥–µ–ª—å –∏ –æ–ø—Ç–∏–º–∏–∑–∞—Ç–æ—Ä
    num_labels = len(label2id)
    model = AutoModelForSequenceClassification.from_pretrained(
        model_name,
        num_labels=num_labels,
        id2label=id2label,
        label2id=label2id,
    )

    optimizer = torch.optim.AdamW(model.parameters(), lr=lr)

    # 5. –û–±–æ—Ä–∞—á–∏–≤–∞–µ–º –≤—Å—ë –≤ accelerator
    model, optimizer, train_dataloader, val_dataloader = accelerator.prepare(
        model, optimizer, train_dataloader, val_dataloader
    )

    # 6. –¶–∏–∫–ª –æ–±—É—á–µ–Ω–∏—è
    for epoch in range(num_epochs):
        accelerator.print(f"\nEpoch {epoch + 1}/{num_epochs}")
        train_one_epoch(model, optimizer, train_dataloader, accelerator)
        evaluate(model, val_dataloader, accelerator)

    # 7. –°–æ—Ö—Ä–∞–Ω—è–µ–º –º–æ–¥–µ–ª—å (—Ä–∞–∑–≤–æ—Ä–∞—á–∏–≤–∞–µ–º –µ—ë —Å–Ω–∞—á–∞–ª–∞)
    unwrapped_model = accelerator.unwrap_model(model)
    unwrapped_model.save_pretrained("models/tickets_accelerate")
    tokenizer.save_pretrained("models/tickets_accelerate")

    accelerator.print("Training finished, model saved to models/tickets_accelerate")

4.3. –§—É–Ω–∫—Ü–∏—è train_one_epoch

–ó–∞–¥–∞—á–∞:

  • –ø–µ—Ä–µ–≤–æ–¥–∏–º –º–æ–¥–µ–ª—å –≤ train() —Ä–µ–∂–∏–º,
  • –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –±–∞—Ç—á–∞:
    • forward,
    • loss,
    • backward,
    • optimizer.step,
    • optimizer.zero_grad,
  • –ª–æ–≥–∏—Ä—É–µ–º —Å—Ä–µ–¥–Ω–∏–π –ª–æ—Å—Å –ø–æ —ç–ø–æ—Ö–µ.
def train_one_epoch(model, optimizer, train_dataloader, accelerator):
    model.train()
    total_loss = 0.0

    for step, batch in enumerate(train_dataloader):
        # batch —É–∂–µ –Ω–∞ –Ω—É–∂–Ω–æ–º –¥–µ–≤–∞–π—Å–µ (accelerator.prepare —Ç–∞–∫ —Å–¥–µ–ª–∞–ª)
        outputs = model(
            input_ids=batch["input_ids"],
            attention_mask=batch["attention_mask"],
            labels=batch["labels"],
        )
        loss = outputs.loss

        # backward —á–µ—Ä–µ–∑ accelerator (–æ–Ω —Å–∞–º —Ä–∞–∑—Ä—É–ª–∏—Ç fp16/DDP)
        accelerator.backward(loss)

        optimizer.step()
        optimizer.zero_grad()

        total_loss += loss.item()

        if (step + 1) % 50 == 0:
            avg_loss = total_loss / (step + 1)
            accelerator.print(f"  Step {step + 1}/{len(train_dataloader)} | loss={avg_loss:.4f}")

    avg_epoch_loss = total_loss / len(train_dataloader)
    accelerator.print(f"  Train loss: {avg_epoch_loss:.4f}")

–†–∞–∑–±–æ—Ä:

  • model.train() ‚Üí –≤–∫–ª—é—á–∞–µ—Ç dropout –∏ —Ç.–ø.
  • batch —É–∂–µ –ª–µ–∂–∏—Ç –Ω–∞ –Ω—É–∂–Ω–æ–º device (accelerator —Å–¥–µ–ª–∞–ª).

4.4. –§—É–Ω–∫—Ü–∏—è evaluate

–ó–∞–¥–∞—á–∞:

  • –±–µ–∑ –≥—Ä–∞–¥–∏–µ–Ω—Ç–æ–≤,
  • —Å—á–∏—Ç–∞–µ–º:
    • —Å—Ä–µ–¥–Ω–∏–π eval loss,
    • accuracy –Ω–∞ –≤–∞–ª–∏–¥–∞—Ü–∏–∏.
def evaluate(model, val_dataloader, accelerator):
    model.eval()
    total_loss = 0.0
    correct = 0
    total = 0

    for batch in val_dataloader:
        with torch.no_grad():
            outputs = model(
                input_ids=batch["input_ids"],
                attention_mask=batch["attention_mask"],
                labels=batch["labels"],
            )

        loss = outputs.loss
        logits = outputs.logits

        total_loss += loss.item()

        preds = torch.argmax(logits, dim=-1)
        correct += (preds == batch["labels"]).sum().item()
        total += batch["labels"].size(0)

    avg_loss = total_loss / len(val_dataloader)
    accuracy = correct / total if total > 0 else 0.0

    accelerator.print(f"  Val loss: {avg_loss:.4f} | Val accuracy: {accuracy:.4f}")
    return avg_loss, accuracy

def train_one_epoch(model, optimizer, train_dataloader, accelerator):
    model.train()
    total_loss = 0.0

    for step, batch in enumerate(train_dataloader):
        with accelerator.accumulate(model):
            outputs = model(
                input_ids=batch["input_ids"],
                attention_mask=batch["attention_mask"],
                labels=batch["labels"],
            )
            loss = outputs.loss

            accelerator.backward(loss)
            optimizer.step()
            optimizer.zero_grad()

        total_loss += loss.item()

        if (step + 1) % 50 == 0:
            avg_loss = total_loss / (step + 1)
            accelerator.print(
                f"  Step {step + 1}/{len(train_dataloader)} | "
                f"loss={avg_loss:.4f}"
            )

    avg_epoch_loss = total_loss / len(train_dataloader)
    accelerator.print(f"  Train loss: {avg_epoch_loss:.4f}")

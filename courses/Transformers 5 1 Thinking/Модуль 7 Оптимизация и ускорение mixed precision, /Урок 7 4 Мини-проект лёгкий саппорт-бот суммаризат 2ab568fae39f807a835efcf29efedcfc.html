def classify_ticket(text: str):
    """
    –ü—Ä–∏–Ω–∏–º–∞–µ—Ç —Ç–µ–∫—Å—Ç —Ç–∏–∫–µ—Ç–∞ –∏–ª–∏ –¥–∏–∞–ª–æ–≥–∞,
    –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∫–ª–∞—Å—Å —Ç–∏–∫–µ—Ç–∞ –∏ —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å.
    """
    enc = cls_tokenizer(
        text,
        return_tensors="pt",
        truncation=True,
        max_length=128,
    ).to(DEVICE)

    with torch.no_grad():
        outputs = cls_model(**enc)
        logits = outputs.logits

    probs = torch.softmax(logits, dim=-1)[0]
    pred_id = torch.argmax(probs).item()
    id2label = cls_model.config.id2label

    return {
        "label": id2label[pred_id],
        "confidence": float(probs[pred_id].item()),
    }

def summarize_dialog(dialog: str, max_new_tokens: int = 64) -> str:
    """
    –î–µ–ª–∞–µ—Ç –∫—Ä–∞—Ç–∫–æ–µ —Ä–µ–∑—é–º–µ –¥–∏–∞–ª–æ–≥–∞.
    """
    enc = summ_tokenizer(
        dialog,
        return_tensors="pt",
        truncation=True,
        max_length=512,  # –º–æ–∂–Ω–æ –∞–¥–∞–ø—Ç–∏—Ä–æ–≤–∞—Ç—å –ø–æ–¥ –¥–ª–∏–Ω—É –¥–∏–∞–ª–æ–≥–æ–≤
    ).to(DEVICE)

    with torch.no_grad():
        out_ids = summ_model.generate(
            **enc,
            max_new_tokens=max_new_tokens,
            num_beams=4,
            no_repeat_ngram_size=3,
        )

    summary = summ_tokenizer.decode(out_ids[0], skip_special_tokens=True)
    return summary.strip()

def build_reply_prompt(dialog: str, summary: str) -> str:
    """
    –°—Ç—Ä–æ–∏–º –ø—Ä–æ–º–ø—Ç –¥–ª—è Causal LM-–±–æ—Ç–∞.
    –ú–æ–∂–Ω–æ —É—Å–ª–æ–∂–Ω—è—Ç—å –ø–æ–¥ —Å–µ–±—è (—è–∑—ã–∫, —Å—Ç–∏–ª—å, –¥–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ –∏–Ω—Å—Ç—Ä—É–∫—Ü–∏–∏).
    """
    prompt = (

def pick_emotions(raw_scores, threshold=0.3, top_k=None):
    """
    raw_scores: —Å–ø–∏—Å–æ–∫ —Å–ª–æ–≤–∞—Ä–µ–π {"label": ..., "score": ...}
    threshold: –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–π score, —á—Ç–æ–±—ã –≤—ã–±—Ä–∞—Ç—å —ç–º–æ—Ü–∏—é
    top_k: –µ—Å–ª–∏ –∑–∞–¥–∞–Ω–æ, –±–µ—Ä—ë–º –Ω–µ –±–æ–ª—å—à–µ top_k —ç–º–æ—Ü–∏–π —Å –Ω–∞–∏–±–æ–ª—å—à–∏–º score
    """
    # –æ—Ç—Å–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ —É–±—ã–≤–∞–Ω–∏—é score
    sorted_scores = sorted(raw_scores, key=lambda x: x["score"], reverse=True)

    if top_k is not None:
        sorted_scores = sorted_scores[:top_k]

    picked = [d for d in sorted_scores if d["score"] >= threshold]
    return picked

def pick_emotions(raw_scores, threshold=0.3, top_k=5):
    sorted_scores = sorted(raw_scores, key=lambda x: x["score"], reverse=True)
    if top_k is not None:
        sorted_scores = sorted_scores[:top_k]
    picked = [d for d in sorted_scores if d["score"] >= threshold]
    return picked

def classify_tickets(tickets: list[str]) -> list[dict]:
    """
    –ù–∞ –≤—Ö–æ–¥: —Å–ø–∏—Å–æ–∫ —Ç–µ–∫—Å—Ç–æ–≤ –æ–±—Ä–∞—â–µ–Ω–∏–π
    –ù–∞ –≤—ã—Ö–æ–¥: —Å–ø–∏—Å–æ–∫ —Å–ª–æ–≤–∞—Ä–µ–π:
    {
      "text": ...,
      "sentiment": {"label": ..., "score": ...},
      "emotions": [{"label": ..., "score": ...}, ...]
    }
    """
    # 1) –¢–æ–Ω–∞–ª—å–Ω–æ—Å—Ç—å
    sentiment_preds = sentiment_clf(tickets, batch_size=16)

    # 2) –≠–º–æ—Ü–∏–∏
    emotion_raw = emotion_clf(tickets, batch_size=8)

    results = []
    for text, sent, emo_raw in zip(tickets, sentiment_preds, emotion_raw):
        emotions = pick_emotions(emo_raw, threshold=0.3, top_k=5)
        results.append({
            "text": text,
            "sentiment": {
                "label": sent["label"],
                "score": float(sent["score"]),
            },
            "emotions": [
                {"label": e["label"], "score": float(e["score"])}
                for e in emotions
            ]
        })
    return results

def answer_from_faq(
    question: str,
    faqs: list[dict],
    qa_pipe,
    min_score: float = 0.1,
):
    """
    question: –≤–æ–ø—Ä–æ—Å –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    faqs: —Å–ø–∏—Å–æ–∫ —Å–ª–æ–≤–∞—Ä–µ–π {"title": ..., "text": ...}
    qa_pipe: pipeline("question-answering")
    min_score: –º–∏–Ω–∏–º–∞–ª—å–Ω–∞—è —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å, –Ω–∏–∂–µ –∫–æ—Ç–æ—Ä–æ–π –º—ã –≥–æ–≤–æ—Ä–∏–º "–Ω–µ —É–≤–µ—Ä–µ–Ω"
    """
    candidates = []

    for faq in faqs:
        context = faq["text"]
        res = qa_pipe(question=question, context=context)
        candidates.append({
            "title": faq["title"],
            "answer": res["answer"],
            "score": res["score"],
        })

    # —Å–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç–∏
    candidates = sorted(candidates, key=lambda x: x["score"], reverse=True)
    best = candidates[0]

    if best["score"] < min_score:
        # –º–æ–¥–µ–ª—å –Ω–µ —É–≤–µ—Ä–µ–Ω–∞ –Ω–∏ –≤ –æ–¥–Ω–æ–º –∫–æ–Ω—Ç–µ–∫—Å—Ç–µ
        return {
            "answer": None,
            "source": None,
            "score": best["score"],
            "candidates": candidates,
        }

    return {
        "answer": best["answer"],
        "source": best["title"],
        "score": best["score"],
        "candidates": candidates,
    }

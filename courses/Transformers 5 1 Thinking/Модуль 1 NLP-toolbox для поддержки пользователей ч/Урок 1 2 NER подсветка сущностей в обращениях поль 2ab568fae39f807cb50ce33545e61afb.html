def highlight_entities(text: str, entities: list[dict]) -> str:
    """
    text: –∏—Å—Ö–æ–¥–Ω—ã–π —Ç–µ–∫—Å—Ç
    entities: —Å–ø–∏—Å–æ–∫ —Å–ª–æ–≤–∞—Ä–µ–π –∏–∑ NER-–ø–∞–π–ø–ª–∞–π–Ω–∞ (—Å start, end, entity_group)
    return: —Å—Ç—Ä–æ–∫–∞ —Å –ø–æ–¥—Å–≤–µ—Ç–∫–æ–π –≤–∏–¥–∞ [—Å–ª–æ–≤–æ<TYPE>]
    """
    # —Å–æ—Ä—Ç–∏—Ä—É–µ–º —Å—É—â–Ω–æ—Å—Ç–∏ –ø–æ –ø–æ–∑–∏—Ü–∏–∏ –≤ —Ç–µ–∫—Å—Ç–µ
    entities_sorted = sorted(entities, key=lambda x: x["start"])

    result = []
    last_idx = 0

    for e in entities_sorted:
        start = e["start"]
        end = e["end"]
        label = e.get("entity_group") or e.get("entity")

        # –¥–æ–±–∞–≤–ª—è–µ–º —Ç–µ–∫—Å—Ç –¥–æ —Å—É—â–Ω–æ—Å—Ç–∏

        # –¥–æ–±–∞–≤–ª—è–µ–º –ø–æ–¥—Å–≤–µ—á–µ–Ω–Ω—É—é —Å—É—â–Ω–æ—Å—Ç—å

def highlight_entities_html(text: str, entities: list[dict]) -> str:
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç HTML-—Å—Ç—Ä–æ–∫—É —Å <span> –≤–æ–∫—Ä—É–≥ —Å—É—â–Ω–æ—Å—Ç–µ–π.

    entities_sorted = sorted(entities, key=lambda x: x["start"])
    result = []
    last_idx = 0

    for e in entities_sorted:
        start = e["start"]
        end = e["end"]
        label = e.get("entity_group") or e.get("entity")

        result.append(text[last_idx:start])

        # –ø—Ä–æ—Å—Ç–∞—è –ø–æ–¥—Å–≤–µ—Ç–∫–∞: —Ü–≤–µ—Ç –ø–æ —Ç–∏–ø—É —Å—É—â–Ω–æ—Å—Ç–∏

        span = f'<span style="background-color: {color};" title="{label}">{text[start:end]}</span>'
        result.append(span)

        last_idx = end

    result.append(text[last_idx:])
    return "".join(result)
